# Sales & Expense Management System - Requirements
# نظام إدارة المبيعات والمصروفات - متطلبات النظام

# ============================================================================
# SYSTEM REQUIREMENTS / متطلبات النظام

[System]
# Operating Systems / أنظمة التشغيل
Windows >= 10.0
Linux >= Ubuntu 18.04 LTS (or equivalent)
macOS >= 10.14 (Mojave)

# Hardware Requirements / متطلبات الأجهزة

Storage >= 50MB free disk space
Processor: Any modern x86/x64 CPU

# Terminal Requirements / متطلبات المحطة الطرفية
Terminal: UTF-8 compatible terminal emulator
Encoding: UTF-8 support for Arabic text display

Standard: C++11 or higher

# ============================================================================
# TROUBLESHOOTING / استكشاف الأخطاء وإصلاحها
# ============================================================================

[Common Issues]
# Compilation Errors / أخطاء التجميع
"C++11 not supported": Upgrade compiler or use -std=c++11
"CMake not found": Install CMake from package manager
"Permission denied": chmod +x sales_system

# Runtime Errors / أخطاء وقت التشغيل
"Cannot create file": Check write permissions in directory
"Arabic text garbled": Set terminal encoding to UTF-8
"Segmentation fault": Usually indicates data corruption in sales_data.json

# Performance Issues / مشاكل الأداء
"Slow startup": Compile with -O2 or -O3 optimization
"Large memory usage": Clear old data or increase system RAM
